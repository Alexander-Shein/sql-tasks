Create a query to return a list ordered by order id descending:

| Order Id | Order Total Price | Total Order Discount, % |

SELECT
	o.orderid,
	SUM( od.unitprice*od.quantity ) AS totalprice,
	(
		SUM( od.unitprice*od.quantity + od.unitprice + od.quantity*od.discount )- SUM( od.unitprice*od.quantity )
	)/ SUM( od.unitprice*od.quantity + od.unitprice + od.quantity*od.discount ) AS total_order_discount,
FROM
	orders o
JOIN order_details od ON
	o.orderid = od.orderid
GROUP BY
	o.orderid
ORDER BY
	orderid DESC;
