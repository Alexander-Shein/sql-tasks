Create a SQL query to return the total number of orders for every month in 1997 year:

| January | February | March | April | May | June | July | August | September | November | December |
___________

SELECT COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 1 THEN 1 END) AS [January],
COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 2 THEN 1 END) AS [February],
COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 3 THEN 1 END) AS [March],
COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 4 THEN 1 END) AS [April],
COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 5 THEN 1 END) AS [May],
COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 6 THEN 1 END) AS [June],
COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 7 THEN 1 END) AS [July],
COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 8 THEN 1 END) AS [August],
COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 9 THEN 1 END) AS [September],
COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 10 THEN 1 END) AS [November],
COUNT(CASE WHEN datepart(MONTH, [o].[OrderDate]) = 11 THEN 1 END) AS [December]
FROM [Northwind].[dbo].[Orders] [o]
WHERE datepart(YEAR, [o].[OrderDate]) = '1997'
