Create a SQL query to return the total number of orders for every month in 1997 year:

| January | February | March | April | May | June | July | August | September | November | December |

SELECT COUNT(CASE WHEN DATEPART(MM, OrderDate) = 01 then 1 ELSE NULL END) AS January,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 02 then 1 ELSE NULL END) AS February,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 03 then 1 ELSE NULL END) AS March,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 04 then 1 ELSE NULL END) AS April,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 05 then 1 ELSE NULL END) AS May,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 06 then 1 ELSE NULL END) AS June,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 07 then 1 ELSE NULL END) AS July,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 08 then 1 ELSE NULL END) AS August,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 09 then 1 ELSE NULL END) AS September,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 10 then 1 ELSE NULL END) AS October,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 11 then 1 ELSE NULL END) AS November,
	   COUNT(CASE WHEN DATEPART(MM, OrderDate) = 12 then 1 ELSE NULL END) AS December

FROM Northwind.dbo.Orders
