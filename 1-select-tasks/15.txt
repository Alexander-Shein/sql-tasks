Create a SQL query to return the total number of orders for every month in 1997 year:

| January | February | March | April | May | June | July | August | September | November | December |

SELECT 
SUM (CASE  WHEN MONTH(o.OrderDate)=1 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS January,
SUM (CASE  WHEN MONTH(o.OrderDate)=2 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS February,
SUM (CASE  WHEN MONTH(o.OrderDate)=3 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS March,
SUM (CASE  WHEN MONTH(o.OrderDate)=4 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS April,
SUM (CASE  WHEN MONTH(o.OrderDate)=5 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS May,
SUM (CASE  WHEN MONTH(o.OrderDate)=6 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS June,
SUM (CASE  WHEN MONTH(o.OrderDate)=7 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS July,
SUM (CASE  WHEN MONTH(o.OrderDate)=8 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS August,
SUM (CASE  WHEN MONTH(o.OrderDate)=9 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS September,
SUM (CASE  WHEN MONTH(o.OrderDate)=10 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS October,
SUM (CASE  WHEN MONTH(o.OrderDate)=11 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS November,
SUM (CASE  WHEN MONTH(o.OrderDate)=12 AND YEAR(o.OrderDate)=1997 THEN 1 ELSE 0 END ) AS December
FROM dbo.Orders o;
