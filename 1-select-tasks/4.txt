SELECT [o].[CustomerID], COUNT([o].[CustomerID]) AS [Total number of Orders],
(SELECT COUNT([o].[OrderID])*100.0/COUNT([ord].[OrderID])
FROM [dbo].[Orders] [ord]) AS [ % of all orders ] 
FROM [dbo].[Orders] [o]
GROUP BY [o].CustomerID
ORDER BY [ % of all orders ] DESC
