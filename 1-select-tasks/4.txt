--4Create a query to return:

--| Customer Id | Total number of Orders | % of all orders |
 
--order data by % - higher percent at the top.

SELECT
	[o].[CustomerID] AS [Customer Id],
	COUNT([o].[OrderID]) AS [Total number of Orders],
	ROUND(100.0 * COUNT([o].[OrderId]) / (SELECT COUNT([o].[OrderId]) FROM [Orders][o]), 3) AS [% of all orders]
FROM[dbo].[Orders][o]
GROUP BY [o].[CustomerID]
ORDER BY [% of all orders] DESC
