'SELECT od.OrderID,
SUM(od.Quantity*od.UnitPrice) as OrderTotalPrice, 
(SUM(od.Discount*od.Quantity))/(SUM(od.Quantity*od.UnitPrice))*100 as TotalOrderDiscount 
FROM order_details od
 GROUP BY od.OrderID;
